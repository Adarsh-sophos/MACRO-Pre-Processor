<# 	Here same macro ‘sum’ is defined 
	multiple times 	with different number of parameters
	#>

$macd ...
SUM(a,b,c=1,d=2,e=3)
print("This SUM has 5 parameters : "+ a + b + c + d + e);
$$

$macd ...
SUM()
print("This sum has no parameters");
$$

$macd ...
SUM(a,b,c)
print("This sum has 3 parameters :" + a + b + c);
$$

SUM(10,20,30,40)
SUM()
SUM(10,20,30)
